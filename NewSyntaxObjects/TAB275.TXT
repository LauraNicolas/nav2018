    LOCAL PROCEDURE GetCurrencyCode@1() : Code[10];
    VAR
      BankAcc@1000 : Record "Bank Account";
    BEGIN
      if "Bank Account No." = BankAcc."No." then
        exit(BankAcc."Currency Code");

      if BankAcc.GET("Bank Account No.") then
        exit(BankAcc."Currency Code");

      exit('');
    END;

    BEGIN
    END.
