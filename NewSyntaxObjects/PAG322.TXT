                                   exit;

                                   if VATEntry.GET(ECSLVATReportLineRelation."VAT Entry No.") then begin
                                     TempVATEntry.TRANSFERFIELDS(VATEntry,true);
                                     TempVATEntry.INSERT;
                                   end;
                                 until ECSLVATReportLineRelation.NEXT = 0;

                                 VATEntriesPreview.RUN;
                               END;
