    PROCEDURE UpdateNo@5();
    VAR
      Item@1001 : Record Item;
      GLAccount@1000 : Record "G/L Account";
    BEGIN
      case Type of
        Type::Item:
          begin
            Item.SETRANGE(Id,"Item Id");
            if not Item.FINDFIRST then
              exit;

            "No." := Item."No.";
          end;
        Type::"G/L Account":
          begin
            GLAccount.SETRANGE(Id,"Account Id");
            if not GLAccount.FINDFIRST then
              exit;

            "No." := GLAccount."No.";
          end;
      end;
    END;
