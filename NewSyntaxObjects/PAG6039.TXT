      TempServCrMemoLine.COPYFILTERS(Rec);
      TempServCrMemoLine.SETRANGE("Document No.","Document No.");

        TempServCrMemoLine := ServCrMemoLine;
        TempServCrMemoLine.INSERT;
      end;

    END;
