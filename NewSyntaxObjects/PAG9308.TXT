    LOCAL PROCEDURE SetControlAppearance@5();
    VAR
      ApprovalsMgmt@1000 : Codeunit "Approvals Mgmt.";
      WorkflowWebhookManagement@1001 : Codeunit "Workflow Webhook Management";
    BEGIN
      OpenApprovalEntriesExist := ApprovalsMgmt.HasOpenApprovalEntries(RECORDID);

      CanCancelApprovalForRecord := ApprovalsMgmt.CanCancelApprovalForRecord(RECORDID);

      WorkflowWebhookManagement.GetCanRequestAndCanCancel(RECORDID,CanRequestApprovalForFlow,CanCancelApprovalForFlow);
    END;

    LOCAL PROCEDURE Post@4(PostingCodeunitID@1000 : Integer);
    VAR
      ApplicationAreaSetup@1002 : Record "Application Area Setup";
      LinesInstructionMgt@1001 : Codeunit "Lines Instruction Mgt.";
    BEGIN
      if ApplicationAreaSetup.IsFoundationEnabled then
        LinesInstructionMgt.PurchaseCheckAllLinesHaveQuantityAssigned(Rec);

      SendToPosting(PostingCodeunitID);

      if ApplicationAreaSetup.IsFoundationEnabled then
        ShowPostedConfirmationMessage;
    END;

    LOCAL PROCEDURE ShowPostedConfirmationMessage@7();
    VAR
      PurchInvHeader@1000 : Record "Purch. Inv. Header";
    BEGIN
      PurchInvHeader.SETFILTER("Pre-Assigned No.","No.");
      if PurchInvHeader.FINDFIRST then
        if DIALOG.CONFIRM(OpenPostedPurchaseInvQst,false) then
          PAGE.RUN(PAGE::"Posted Purchase Invoice",PurchInvHeader);
    END;

    LOCAL PROCEDURE VerifyTotal@1(PurchaseHeader@1000 : Record "Purchase Header");
    BEGIN
      if not PurchaseHeader.IsTotalValid then
        ERROR(TotalsMismatchErr);
    END;

    BEGIN
    END.
