                    if FIND('-') then
                      repeat
                        VALIDATE("Row Ref. No.",RowRefNo);
                        MODIFY;

                      until NEXT = 0;
                END;

                   MESSAGE(Text000);
                 END;

                  SourceExpr=RowRefNo }

    }
  }

  }
