                 Date.SETRANGE("Period Start",DMY2DATE(1,1,YearFilter),DMY2DATE(30,12,YearFilter));
                 Date.FINDSET;
                 repeat
                   TRANSFERFIELDS(Date);
                   INSERT;
                 until Date.NEXT = 0;
                 FINDFIRST;
               END;
