  OBJECT-PROPERTIES
  {
    Date=20171006D;
    Time=120000T;
    Version List=NAVW111.0;
  }
  PROPERTIES
  {
    Editable=false;
    CaptionML=ENU=Worksheet Names List;
    SourceTable="Whse. Worksheet Name";
    PageType=List;
    OnInit=BEGIN
             SETRANGE("Worksheet Template Name");
           END;

    OnOpenPage=BEGIN
                 WhseWkshLine.OpenWhseWkshBatch(Rec);
               END;

    OnFindRecord=BEGIN
                   if FIND(Which) then begin
                     PickWkshName := Rec;
                     while true do begin
                       if WMSMgt.LocationIsAllowed("Location Code") then
                         exit(true);
                       if NEXT(1) = 0 then begin
                         Rec := PickWkshName;
                         if FIND(Which) then
                           while true do begin
                             if WMSMgt.LocationIsAllowed("Location Code") then
                               exit(true);
                             if NEXT(-1) = 0 then
                               exit(false);
                           end;
                       end;
                     end;
                   end;
                   exit(false);
                 END;

    OnNextRecord=VAR
                   RealSteps@1001 : Integer;
                   NextSteps@1000 : Integer;
                 BEGIN
                   if Steps = 0 then
                     exit;

                   PickWkshName := Rec;
                   repeat
                     NextSteps := NEXT(Steps / ABS(Steps));
                     if WMSMgt.LocationIsAllowed("Location Code") then begin
                       RealSteps := RealSteps + NextSteps;
                       PickWkshName := Rec;
                     end;
                   until (NextSteps = 0) or (RealSteps = Steps);
                   Rec := PickWkshName;
                   FIND;
                   exit(RealSteps);
                 END;

    ActionList=ACTIONS
    {
      { 1900000004;0 ;ActionContainer;
                      Name=ActionContainer1900000004;
                      ActionContainerType=ActionItems }
      { 11      ;1   ;Action    ;
                      Name=Edit Worksheet;

                      OnAction=BEGIN

                               END;
                                }
    }
