      CustInvoiceDisc@1001 : Record "Cust. Invoice Disc.";
    BEGIN
      CustInvoiceDisc.SETRANGE(Code,CustomerCode);
      if CustInvoiceDisc.FINDSET then
        repeat
          INIT;
          Code := CustInvoiceDisc.Code;
