    LOCAL PROCEDURE SetUsageFilter@1(ModifyRec@1000 : Boolean);
    BEGIN
      if ModifyRec then
        if MODIFY then;
      FILTERGROUP(2);
      case ReportUsage2 of

          SETRANGE(Usage,Usage::Reminder);
        ReportUsage2::"Fin. Charge":
          SETRANGE(Usage,Usage::"Fin.Charge");

          SETRANGE(Usage,Usage::"F.C.Test");
      end;
      FILTERGROUP(0);
      CurrPage.UPDATE;
