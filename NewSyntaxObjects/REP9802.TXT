                                  NewPermissionSet.INIT;

                                  NewPermissionSet.INSERT;

                                    repeat
                                      NewPermission.INIT;
                                      NewPermission.COPY(Permission);
                                      NewPermission."Role ID" := NewRoleId;
                                      NewPermission.INSERT;

                                END;
