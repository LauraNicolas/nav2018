    OnFindRecord=VAR
                   Found@1000 : Boolean;
                 BEGIN
                   if RunOnTempRec then begin
                     TempFilteredItem := Rec;
                     Found := TempFilteredItem.FIND(Which);
                     if Found then
                       Rec := TempFilteredItem;
                     exit(Found);
                   end;
                   exit(FIND(Which));
                 END;

                   ResultSteps@1000 : Integer;
                 BEGIN
                   if RunOnTempRec then begin
                     TempFilteredItem := Rec;

                     if ResultSteps <> 0 then
                       Rec := TempFilteredItem;
                     exit(ResultSteps);
                   end;
                   exit(NEXT(Steps));
                 END;
