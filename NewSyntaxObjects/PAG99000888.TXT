    LOCAL PROCEDURE SetDateFilter@4();
    BEGIN
      if AmountType = AmountType::"Net Change" then
        WorkCenter.SETRANGE("Date Filter","Period Start","Period End")

        WorkCenter.SETRANGE("Date Filter",0D,"Period End");
    END;

    END.
