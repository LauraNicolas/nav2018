    LOCAL PROCEDURE CheckGLAcc@2(AccNo@1000 : Code[20]);
    VAR
      GLAcc@1001 : Record "G/L Account";
    BEGIN
      if AccNo <> '' then begin
        GLAcc.GET(AccNo);
        GLAcc.CheckGLAcc;
        GLAcc.TESTFIELD("Direct Posting",true);
      end;
    END;

    PROCEDURE GetDescriptionInCurrentLanguage@1() : Text[100];
    VAR
      Language@1000 : Record Language;
      PaymentMethodTranslation@1001 : Record "Payment Method Translation";
    BEGIN
      Language.SETRANGE("Windows Language ID",GLOBALLANGUAGE);
      if not Language.FINDFIRST then
        exit(Description);

      if PaymentMethodTranslation.GET(Code,Language.Code) then
        exit(PaymentMethodTranslation.Description);

      exit(Description);
    END;

    BEGIN
    END.
